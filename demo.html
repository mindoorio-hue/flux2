Endpoint

Headers

Authorization: Bearer YOUR_API_KEY
Content-Type: application/json

Common Parameters

// All workflows support these parameters
{
  "prompt": "required - Text description",
  "negative_prompt": "optional - What to avoid",
  "width": 1024,  // Multiple of 8, default: 1024
  "height": 1024, // Multiple of 8, default: 1024
  "num_inference_steps": 50, // dev: 50, schnell: 4
  "guidance_scale": 7.5,
  "num_images": 1, // 1-4 images
  "seed": 42,  // Optional, for reproducibility
  "output_format": "PNG" // PNG, JPEG, or WEBP
}
